#1 para correr en el navegador
uvicorn api(nombre del archivo):app 
uvicorn api:app 

#2 para elegir en que puerto levantarlo
uvicorn api:app -- port 8001 (en caso de que el 8000 esté ocupado por ejemplo)

#3 para recargar la página cuando realizamos cambios a nivel de archivo
también nos sirve para ver en el navegador constantemente los cambios que realizamos
sin necesidad de detener y volver a correr el servicio
uvicorn api:app -- reload

#4 Las funciones --port -- reload son combinables
